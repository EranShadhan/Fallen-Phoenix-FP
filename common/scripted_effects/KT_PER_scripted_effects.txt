PER_improve_army = {
	if = {
		limit = {
			has_idea = PER_army_1_focus
		}
		swap_ideas = {
			remove_idea = PER_army_1_focus
			add_idea = PER_army_2_focus
		}
		else_if = {
			limit = {
				has_idea = PER_army_2_focus
			}
			swap_ideas = {
				remove_idea = PER_army_2_focus
				add_idea = PER_army_3_focus
			}
		}
		else_if = {
			limit = {
				has_idea = PER_army_3_focus
			}
			swap_ideas = {
				remove_idea = PER_army_3_focus
				add_idea = PER_army_4_focus
			}
		}
		else_if = {
			limit = {
				has_idea = PER_army_4_focus
			}
			swap_ideas = {
				remove_idea = PER_army_4_focus
				add_idea = PER_army_5_focus
			}
		}
		else = {
			remove_ideas = PER_army_5_focus
		}
	}
}
IRA_push_germany = {
	custom_effect_tooltip = IRA_germany_not_happy
	if = {
		limit = { 
			OR = {
				check_variable = { IRA_push_germany = 0 }
				has_country_flag = {
					flag = IRA_push_germany
					days < 100
				}
			}
		}
		add_to_variable = { IRA_push_germany = 1 }
	}
	clr_country_flag = IRA_push_germany
	hidden_effect = {
		if = {
			limit = { check_variable = { IRA_push_germany > 3 } }
			#germany does something
		}
		else = {
			set_country_flag = IRA_push_germany
		}
	}
}
